 flexbox uses

 flex-container{
   display: flex; / works as a Row/column
   flex-direction: column; / Whether Row or column
   justify-content: space-around; / align main axis
   align-items: center; / align cross axis
   flex-wrap: wrap; / works as a Wrap
   align-content: center; / aligning when wrap
 }


flex-item{
  flex-grow: 1; / works as a Expanded
  flex-shrink: 0; / ignore flexiblity
  flex-basis: 0; / grow space 
  align-self: center; / override align-items 
  order : 1; changing order of items

}
